\begin{abstract}
基于 Phase~1 的分析结论，本文正式采用 per-prompt-set 扩展的 MoF 架构：logits $\ell \in \R^{K \times T \times S}$，其中 $K$ = teacher 数量，$T$ = 推理 timesteps，$S$ = prompt set 数量。每个 prompt set 拥有独立的 teacher 混合策略。本文给出完整的 loss 推导、梯度公式、收敛性分析，并详细设计 per-set reward 策略——包括 reward normalization、advantage computation、OOD bonus 的精确公式和实现细节。
\end{abstract}

\subsection{Per-Set Velocity Field}

\begin{definition}[Per-Set Combined Velocity]
对 prompt $p \in \mathcal{P}_s$（属于 prompt set $s$），定义：
\begin{equation}
\boxed{
v_{\lambda_s}(x, t_i) = \sum_{k=1}^K \lambda_{k,i,s}\, v^{(k)}(x, t_i), \quad
\lambda_{k,i,s} = \frac{\exp(\ell_{k,i,s}/\tau)}{\sum_{k'=1}^K \exp(\ell_{k',i,s}/\tau)}
}
\label{eq:per_set_velocity}
\end{equation}
其中权重满足 $\lambda_{k,i,s} \geq 0$, $\sum_{k=1}^K \lambda_{k,i,s} = 1$, $\forall\, i, s$。
\end{definition}

\begin{remark}[与 Phase~1 Global $\lambda$ 的关系]
Global $\lambda$ 是 per-set $\lambda$ 的特例：当 $\ell_{k,i,s} = \ell_{k,i}$ 对所有 $s$ 相同时，退化为 Phase~1 的 global 版本。Per-set 扩展引入了 $S$ 倍的参数，但仍然极小（$K \times T \times S = 3 \times 28 \times 3 = 252$ 参数）。
\end{remark}

\subsection{参数空间几何}

\begin{proposition}[搜索空间结构]
Per-set MoF 的搜索空间为 $S$ 个独立的乘积 simplex：
\begin{equation}
\Omega = \prod_{s=1}^S \prod_{i=1}^T \Delta^K = \left(\Delta^K\right)^{T \times S}
\end{equation}
其中 $\Delta^K = \{\lambda \in \R^K_+ : \sum_k \lambda_k = 1\}$ 是 $K$-simplex。

\textbf{关键性质}：不同 prompt set $s$ 和 $s'$ 的参数 $\ell_{:,:,s}$ 和 $\ell_{:,:,s'}$ 完全独立——梯度不交叉，优化完全解耦。
\end{proposition}

\subsection{Inference 流程}

\begin{algorithm}[H]
\caption{Per-Set MoF Inference}
\label{alg:inference}
\begin{algorithmic}[1]
\Require Prompt $p$, noise $x_T \sim \N(0, I)$, logits $\ell \in \R^{K \times T \times S}$
\Ensure Generated image $x_0$
\State Determine prompt set: $s \gets \mathrm{classify}(p)$
\State Compute weights: $\lambda_{:,:,s} \gets \softmax_k(\ell_{:,:,s} / \tau)$ \Comment{shape $(K, T)$}
\State $x \gets x_T$
\For{$i = 1, \ldots, T$}
    \For{$k = 1, \ldots, K$}
        \State $v^{(k)}_i \gets v^{(k)}(x, t_i)$ \Comment{Frozen teacher forward}
    \EndFor
    \State $v_{\mathrm{combined}} \gets \sum_k \lambda_{k,i,s} \cdot v^{(k)}_i$ \Comment{Weighted combination}
    \State $x \gets \mathrm{scheduler\_step}(x, v_{\mathrm{combined}}, t_i)$ \Comment{ODE step}
\EndFor
\State \Return $x$
\end{algorithmic}
\end{algorithm}

\section{Loss 函数}
\label{sec:loss}

\subsection{Per-Set DiffusionNFT Loss}

对每个 prompt set $s$ 独立定义 NFT loss。设 batch 中 prompt $p \in \mathcal{P}_s$：

\begin{definition}[Per-Set NFT Loss (single timestep)]
\begin{equation}
\boxed{
\loss_{i,s}(\ell_{:,:,s}) = \frac{1}{\beta}\left[
r_s \cdot L^+_{i,s} + (1 - r_s) \cdot L^-_{i,s}
\right]
}
\label{eq:per_set_nft_loss}
\end{equation}
其中：
\begin{align}
r_s &= \frac{\adv_s + A_{\max}}{2 A_{\max}} \in [0, 1] \label{eq:normalized_adv} \\
v^{\mathrm{new}}_{i,s} &= \sum_k \lambda_{k,i,s}\, v^{(k)}(x_{t_i}, t_i) \quad\text{(current policy for set $s$)} \label{eq:v_new_s} \\
v^{\mathrm{old}}_{i,s} &= \sum_k \bar{\lambda}_{k,i,s}\, v^{(k)}(x_{t_i}, t_i) \quad\text{(EMA policy for set $s$)} \label{eq:v_old_s} \\
v^+_{i,s} &= \beta\, v^{\mathrm{new}}_{i,s} + (1-\beta)\, v^{\mathrm{old}}_{i,s} \label{eq:v_plus} \\
v^-_{i,s} &= (1+\beta)\, v^{\mathrm{old}}_{i,s} - \beta\, v^{\mathrm{new}}_{i,s} \label{eq:v_minus}
\end{align}
\begin{align}
x^+_{0,s} &= x_{t_i} - \sigma_i\, v^+_{i,s} \label{eq:x0_plus} \\
x^-_{0,s} &= x_{t_i} - \sigma_i\, v^-_{i,s} \label{eq:x0_minus} \\
L^+_{i,s} &= \frac{\|x^+_{0,s} - x_0\|^2}{(w^+_s)^2}, \quad
L^-_{i,s} = \frac{\|x^-_{0,s} - x_0\|^2}{(w^-_s)^2} \label{eq:L_plus_minus}
\end{align}
$w^+_s, w^-_s$ 为 detached mean absolute error（per-sample normalization）。
\end{definition}

\subsection{Full Training Loss}

累积所有 timestep 和 batch samples：
\begin{equation}
\loss_s(\ell_{:,:,s}) = \frac{1}{B_s \cdot T}\sum_{b \in \mathcal{B}_s}\sum_{i=1}^T \loss^{(b)}_{i,s}(\ell_{:,:,s})
\label{eq:full_loss}
\end{equation}
其中 $\mathcal{B}_s$ 为当前 batch 中属于 prompt set $s$ 的 sample 子集，$B_s = |\mathcal{B}_s|$。

\begin{remark}[Per-Set 独立性]
总 loss 为各 set 的 loss 之和：
\begin{equation}
\loss_{\mathrm{total}}(\ell) = \sum_{s=1}^S \loss_s(\ell_{:,:,s})
\end{equation}
由于 $\ell_{:,:,s}$ 只出现在 $\loss_s$ 中，$\frac{\partial \loss_{\mathrm{total}}}{\partial \ell_{:,:,s}} = \frac{\partial \loss_s}{\partial \ell_{:,:,s}}$。各 prompt set 的优化完全独立——如同并行训练 $S$ 个独立的 global-$\lambda$ MoF。
\end{remark}

\section{梯度推导}
\label{sec:gradient}
%% ============================================================

\subsection{Main Theorem}

\begin{theorem}[Per-Set MoF Gradient]
\label{thm:per_set_gradient}
对 prompt set $s$，single sample，single timestep $i$：
\begin{equation}
\boxed{
\frac{\partial \loss_{i,s}}{\partial \ell_{k,i,s}} =
\frac{2}{\tau}\,\lambda_{k,i,s}\left[v^{(k)}_i - v^{\mathrm{new}}_{i,s}\right]^\top
\underbrace{\left[\frac{r_s\,(x^+_{0,s} - x_0)}{(w^+_s)^2} - \frac{(1-r_s)\,(x^-_{0,s} - x_0)}{(w^-_s)^2}\right]}_{\triangleq\; \mathbf{e}_{i,s} \;\in\; \R^d}\cdot(-\sigma_i)
}
\label{eq:per_set_gradient}
\end{equation}

且 $\frac{\partial \loss_{i,s}}{\partial \ell_{k,j,s'}} = 0$ 当 $j \neq i$ 或 $s' \neq s$。
\end{theorem}

\begin{proof}
推导与 Phase~1 Theorem~2.1 完全相同，仅将 global $\lambda_{k,i}$ 替换为 per-set $\lambda_{k,i,s}$。各 set 的 sample 只对自己的 $\ell_{:,:,s}$ 产生梯度，跨 set 梯度为零。
\end{proof}

\subsection{三重解耦性}

\begin{corollary}[Triple Decoupling]
\label{cor:triple_decouple}
Per-set MoF 的梯度具有三重解耦：
\begin{enumerate}[leftmargin=1.5em]
\item \textbf{Timestep 解耦}：$\frac{\partial \loss_{i,s}}{\partial \ell_{k,j,s}} = 0$ for $j \neq i$
\item \textbf{Prompt-set 解耦}：$\frac{\partial \loss_{i,s}}{\partial \ell_{k,i,s'}} = 0$ for $s' \neq s$
\item \textbf{Orthogonal teacher competition}：在每个 $(i, s)$ 固定的 slice 上，$K$ 个 teacher 通过 softmax 竞争。增大 $\ell_{k,i,s}$ 必然减小 $\ell_{k',i,s}$ ($k' \neq k$) 的有效权重。
\end{enumerate}

\textbf{含义}：每个 $(\text{timestep}, \text{prompt set})$ pair 是一个完全独立的 $K$-way 选择问题。优化 252 个参数等价于并行优化 $T \times S = 84$ 个独立的 $K$-simplex 选择。
\end{corollary}

\subsection{Gradient 的分量分析}

将梯度~\eqref{eq:per_set_gradient} 分解为三个语义明确的因子：

\begin{equation}
\frac{\partial \loss_{i,s}}{\partial \ell_{k,i,s}} = \underbrace{\frac{2\sigma_i}{\tau}\,\lambda_{k,i,s}}_{\text{(I) Scale}} \;\cdot\;
\underbrace{\left[v^{(k)}_i - v^{\mathrm{new}}_{i,s}\right]}_{\text{(II) Direction}} \;\cdot\;
\underbrace{(-\mathbf{e}_{i,s})}_{\text{(III) Reward signal}}
\label{eq:grad_decomp}
\end{equation}

\paragraph{Factor (I): Scale.}
\begin{itemize}[leftmargin=1.5em]
\item $\lambda_{k,i,s} \in [0, 1]$：当前权重。权重越接近 0 或 1，梯度越小（softmax 饱和效应）。最大梯度在 $\lambda = 1/K$ 处（均匀分布时，各 teacher 梯度最灵敏）。
\item $\sigma_i$：noise level at timestep $i$。高噪声区域（早期 timestep，$\sigma \approx 1$）梯度较大；低噪声区域（晚期，$\sigma \to 0$）梯度较小——这意味着\textbf{早期 timestep 的权重学习更快}。
\end{itemize}

\paragraph{Factor (II): Direction $[v^{(k)}_i - v^{\mathrm{new}}_{i,s}]$.}
\begin{itemize}[leftmargin=1.5em]
\item Teacher $k$ 相对于当前 combined velocity 的``偏差方向''。
\item 若 $\lambda_{k,i,s} = 1$（已选中 teacher $k$），则 $v^{\mathrm{new}}_{i,s} = v^{(k)}_i$，此项为 0——\textbf{已确定的选择不再变化}。
\item 若 $\lambda_{k,i,s} \ll 1$，则 $v^{(k)}_i - v^{\mathrm{new}}_{i,s} \approx v^{(k)}_i - \sum_{k'\neq k}\lambda_{k'} v^{(k')}_i$——teacher $k$ 与其他 teacher 混合的差异。
\end{itemize}

\paragraph{Factor (III): Reward signal $\mathbf{e}_{i,s}$.}
\begin{itemize}[leftmargin=1.5em]
\item 由 advantage $r_s$ 加权的 prediction error。
\item 当 $r_s > 0.5$（正 advantage，当前 sample 好）：$\mathbf{e}_{i,s}$ 主要由 $L^+$ 方向决定。
\item 当 $r_s < 0.5$（负 advantage，当前 sample 差）：$\mathbf{e}_{i,s}$ 主要由 $-L^-$ 方向决定。
\end{itemize}

\subsection{简化梯度（$\beta = 1$, $w = 1$, $v^{\mathrm{old}} \approx v^{\mathrm{new}}$）}

在标准设定下（$\beta = 1$, 忽略 normalization weight，small off-policy gap）：
\begin{equation}
\boxed{
\frac{\partial \loss_{i,s}}{\partial \ell_{k,i,s}} \approx \frac{2\sigma_i^2}{\tau}\,\lambda_{k,i,s}
\left[v^{(k)}_i - v^{\mathrm{new}}_{i,s}\right]^\top
\left[(2r_s - 1)(v^{\mathrm{new}}_{i,s} - v^*_i)\right]
}
\label{eq:simplified_grad}
\end{equation}
其中 $v^*_i = (x_{t_i} - x_0)/\sigma_i$ 为 ground truth velocity。

\begin{tcolorbox}[colback=green!5, colframe=green!50, title=梯度动力学解读]
设 $\adv_s > 0$（好 sample，$r_s > 0.5$）：
\begin{itemize}[leftmargin=1em]
\item 若 teacher $k$ 的 velocity 比当前混合更接近 ground truth（$[v^{(k)} - v^{\mathrm{new}}]^\top(v^{\mathrm{new}} - v^*) < 0$），则梯度为负 → gradient descent 增大 $\ell_{k,i,s}$ → \textbf{增加 teacher $k$ 在 set $s$, timestep $i$ 上的权重}。
\item 若 teacher $k$ 比混合更差，则权重减小。
\end{itemize}

设 $\adv_s < 0$（差 sample，$r_s < 0.5$）：
\begin{itemize}[leftmargin=1em]
\item 反向信号：惩罚当前混合方向，探索远离当前配置的方向。
\end{itemize}

\textbf{Per-set 的额外优势}：set $s$ 的梯度只看 set $s$ 的 reward 信号——GenEval set 上的 reward 只影响 $\ell_{:,:,\mathrm{gen}}$，OCR set 上的 reward 只影响 $\ell_{:,:,\mathrm{ocr}}$。不同 set 的 reward 信号\textbf{完全互不干扰}。
\end{tcolorbox}

\section{收敛性与最优解分析}
\label{sec:convergence}

\subsection{Per-Set 最优解}

\begin{theorem}[Per-Set 最优解的形式]
\label{thm:optimal}
对 prompt set $s$，optimal logits $\ell^*_{:,:,s}$ 满足：
\begin{equation}
\ell^*_{:,:,s} = \arg\max_{\ell_{:,:,s}} \E_{p \in \mathcal{P}_s}\left[\reward_s\!\left(\mathrm{ODE}(v_{\lambda_s}; x_T, p)\right)\right]
\label{eq:per_set_optimal}
\end{equation}
其中 $\reward_s$ 是 prompt set $s$ 的 reward function（定义见 Section~\ref{sec:reward}）。
\end{theorem}

\subsection{Level~1 保证（$\geq$ Single Teacher）}

\begin{theorem}[Guaranteed Non-Degradation]
\label{thm:non_degrade}
对任意 prompt set $s$ 和 teacher $k$：
\begin{equation}
\max_{\ell_{:,:,s}} \E_{p \in \mathcal{P}_s}\left[\reward_s(v_{\lambda_s})\right] \geq \reward_s(v^{(k)}), \quad \forall k = 1,\ldots,K
\end{equation}

特别地，取 $k = s$（in-domain teacher）：
\begin{equation}
\boxed{
\E[\reward_s(v_{\lambda^*_s})] \geq \E[\reward_s(v^{(s)})] \quad\text{(in-domain performance $\geq$ single teacher)}
}
\end{equation}
\end{theorem}

\begin{proof}
$\lambda_{k,i,s} = \delta_{k,s}$（teacher $s$ 恒取权重 1）是可行解，对应 $v_{\lambda_s} = v^{(s)}$。优化在包含此点的搜索空间上进行，最优解不劣于此。对任意其他 teacher $k$ 同理（$\delta_{k,\cdot}$ 也是可行解）。
\end{proof}

\begin{remark}[与 OPD 的对比]
这是 MoF per-set 架构相比 OPD-Average 的根本优势。OPD-Average 的最优解是 teacher 均值，由 Jensen 不等式必然在 in-domain 上退化。Per-set MoF 则有\textbf{构造性的下界保证}。
\end{remark}

\subsection{Level~2 条件（$>$ Single Teacher）}

\begin{theorem}[Strict Improvement Condition]
\label{thm:strict_improve}
Per-set MoF 严格超越 teacher $s$ on set $s$ 的充分条件为：存在 timestep $i^*$ 和 teacher $k^* \neq s$ 使得：
\begin{equation}
\E_{p \in \mathcal{P}_s}\left[\reward_s\!\left(\mathrm{ODE}(v_{\lambda'}; x_T, p)\right)\right] > \E_{p \in \mathcal{P}_s}\left[\reward_s\!\left(\mathrm{ODE}(v^{(s)}; x_T, p)\right)\right]
\end{equation}
其中 $\lambda'$ 仅在 timestep $i^*$ 处与 $\delta_{k,s}$ 不同：$\lambda'_{k^*,i^*,s} = \epsilon > 0$, $\lambda'_{s,i^*,s} = 1 - \epsilon$。

\textbf{直觉}：只需存在一个 timestep，在该 timestep 上引入少量其他 teacher 的 velocity 能提升 set $s$ 的 reward。
\end{theorem}

\subsection{温度 $\tau$ 与收敛形态}

\begin{proposition}[温度对收敛的影响]
\label{prop:temperature}
\begin{enumerate}[leftmargin=1.5em]
\item $\tau \to 0$（低温极限）：$\lambda_{k,i,s} \to \delta_{k, \arg\max_k \ell_{k,i,s}}$。每个 $(i,s)$ 退化为 hard selection（winner-take-all）。搜索空间退化为 $\{1,\ldots,K\}^{T \times S}$——离散组合优化。

\item $\tau \to \infty$（高温极限）：$\lambda_{k,i,s} \to 1/K$。所有 teacher 均匀混合，等价于 teacher velocity 的算术平均。

\item $\tau$ 适中（推荐 $\tau \in [0.5, 2.0]$）：soft assignment，允许梯度流动的同时保持一定的 specialization tendency。
\end{enumerate}
\end{proposition}

\begin{remark}[温度退火策略]
实践建议：初始 $\tau = 1.0$（充分探索），训练后期逐步降低到 $\tau = 0.3$--$0.5$（promote specialization）。这类似 Gumbel-Softmax 的退火策略。
\end{remark}

\section{Reward 设计（详细版）}
\label{sec:reward}
%% ============================================================

\subsection{Per-Set Composite Reward Definition}

\begin{definition}[Per-Set Composite Reward with OOD Bonus]
\label{def:per_set_reward}
对 prompt $p \in \mathcal{P}_s$ 生成的 image $x$，设 $\mathcal{R}(x, s)$ 为该 sample 的 applicable reward 集合（由 \texttt{applicable\_sources} 过滤，不适用的 reward 得到 NaN 被跳过）：
\begin{equation}
\boxed{
r_{\mathrm{composite}}(x, s) = R_{\mathrm{in}(s)}(x) + \gamma \cdot \frac{1}{|\mathcal{R}_{\mathrm{ood}}|}\sum_{j \in \mathcal{R}_{\mathrm{ood}}} R_j(x)
}
\label{eq:reward_def}
\end{equation}
其中 $\mathrm{in}(s)$ 是 source $s$ 的 in-domain reward（由 \texttt{TeacherConfig.reward\_name} 显式指定），$\mathcal{R}_{\mathrm{ood}} = \mathcal{R}(x,s) \setminus \{\mathrm{in}(s)\}$，$\gamma \geq 0$ 是 OOD bonus 系数。

注意：这里使用的是 \textbf{raw reward}（未做 normalization）。Reward scale 的差异通过下游 advantage 计算中的 group-std normalization 隐式消除。
\end{definition}

\begin{remark}[Reward 权重解读]
\begin{itemize}[leftmargin=1.5em]
\item In-domain reward $R_{\mathrm{in}(s)}$ 的有效系数为 $1$。
\item 每个 OOD reward $R_j$ ($j \neq \mathrm{in}(s)$) 的有效系数为 $\frac{\gamma}{|\mathcal{R}_{\mathrm{ood}}|}$。
\item 当 $\gamma = 0$：纯 in-domain 优化（Level~1 保证最强，但可能收敛到 teacher $s$ 本身）。
\item 当 $\gamma = 0.2$（推荐）：OOD 信号提供探索方向但不干扰主信号。
\end{itemize}
\end{remark}

\subsection{Advantage Computation (GDPO-Style Per-Reward Normalization)}

MoF 采用 GDPO 的核心思想：\textbf{先对每个 reward 独立做 group-level normalization 得到 per-reward advantage，然后在 advantage 空间做 in-domain + OOD 加权聚合}。

\begin{definition}[MoF Advantage Pipeline]
\label{def:advantage}
设 epoch 中有 $N$ 个 sample（跨所有 rank gather 后），按 prompt 分组为 $\mathcal{G}_1, \ldots, \mathcal{G}_M$（每个 group 有 $K$ 个 sample）。

\textbf{Step 1: Per-Reward Group Normalization.} 对每个 reward $k$，独立计算其 advantage：
\begin{equation}
a_k(x_i) = \frac{R_k(x_i) - \mu_{k,g}}{\max(\sigma_{k,g},\, \epsilon)}, \quad
\mu_{k,g} = \frac{1}{|\mathcal{G}_g|}\sum_{j \in \mathcal{G}_g} R_k(x_j), \quad
\sigma_{k,g} = \mathrm{std}(\{R_k(x_j)\}_{j \in \mathcal{G}_g})
\label{eq:per_reward_norm}
\end{equation}
此步将各 reward 统一到 $\sim$ zero-mean, unit-variance 的 advantage 空间，\textbf{消除了 reward scale 差异}（如 GenEval $\in [0,1]$ vs PickScore $\in [0.8, 0.95]$）。

\textbf{Step 2: Per-Set Advantage Aggregation.} 对 sample $x_i$ 来自 source $s$：
\begin{equation}
\boxed{
\mathcal{A}_{\mathrm{combined}}(x_i) = a_{\mathrm{in}(s)}(x_i) + \gamma \cdot \frac{1}{|\mathcal{R}_{\mathrm{ood}}|}\sum_{j \in \mathcal{R}_{\mathrm{ood}}} a_j(x_i)
}
\label{eq:advantage_aggregation}
\end{equation}
其中 $\mathcal{R}_{\mathrm{ood}} = \mathcal{R}(x_i, s) \setminus \{\mathrm{in}(s)\}$ 是适用于该 sample 的非 in-domain reward 集合。

\textbf{Step 3: Batch-Level Normalization.}
\begin{equation}
\mathcal{A}_{\mathrm{final}}(x_i) = \frac{\mathcal{A}_{\mathrm{combined}}(x_i)}{\max(\sigma_{\mathrm{global}},\, \epsilon)}, \quad
\sigma_{\mathrm{global}} = \mathrm{std}(\{\mathcal{A}_{\mathrm{combined}}(x_i)\}_{i=1}^N)
\label{eq:final_norm}
\end{equation}
\end{definition}

\begin{remark}[与原始 GDPO 的关系]
原始 GDPO 的公式为：
\begin{equation}
\mathcal{A}^{\mathrm{GDPO}}_i = \frac{\sum_k w_k \cdot a_k(x_i)}{\sigma_{\mathrm{global}}}
\end{equation}
其中 $w_k$ 是全局静态权重。MoF 的区别在于：\textbf{权重是 per-sample 自适应的}——对来自 source $s$ 的 sample，in-domain reward 权重为 $1$，OOD reward 权重为 $\gamma/|\mathcal{R}_{\mathrm{ood}}|$。这等价于 GDPO with sample-conditional weights。
\end{remark}

\begin{remark}[为什么在 Advantage 空间聚合而非 Raw Reward 空间]
如果在 raw reward 空间聚合（$r = R_{\mathrm{in}} + \gamma \cdot R_{\mathrm{ood}}$），然后做 group normalization，会有以下问题：
\begin{itemize}[leftmargin=1.5em]
\item GenEval reward 范围 $[0, 1]$，PickScore 范围 $[0.8, 0.95]$——scale 不同导致 $\gamma$ 的有效影响力不可预测。
\item 同一 $\gamma = 0.2$ 对不同 reward pair 的实际效果差异巨大。
\end{itemize}
在 advantage 空间聚合（各 reward 已归一化为 $\sim N(0, 1)$）后：
\begin{itemize}[leftmargin=1.5em]
\item $\gamma = 0.2$ 意味着 OOD advantage 贡献 20\% 的信号强度，\textbf{与 reward 原始 scale 无关}。
\item 不需要手动调整 reward weights 来补偿 scale 差异。
\end{itemize}
\end{remark}

\begin{remark}[Cross-Source Group 不混合的 Invariant]
关键约束：同一 group 内的 $K$ 个 sample 必须来自同一 source（因为共享同一 prompt）。因此 per-reward group normalization 中，每个 group 内所有 sample 使用相同的 composite 公式——不会出现 group 内有的 sample 有某个 reward 而有的没有的情况。NaN（非 applicable）只发生在\textbf{跨 group} 维度。
\end{remark}

\subsection{OOD Bonus $\gamma$ 的梯度效应}

\begin{proposition}[OOD Bonus 如何引导探索]
\label{prop:ood_bonus}
设当前 $\lambda^*_{:,:,s} \approx \delta_{k,s}$（已收敛到 in-domain teacher）。引入 $\gamma > 0$ 后，梯度为：
\begin{align}
\frac{\partial \loss_{i,s}}{\partial \ell_{k',i,s}} &\propto \lambda_{k',i,s}\left[v^{(k')}_i - v^{(s)}_i\right]^\top \mathbf{e}_{i,s}
\end{align}
其中 $\mathbf{e}_{i,s}$ 现在包含 OOD reward 的贡献。

当 OOD teacher $k'$ 在某个 timestep $i$ 上能提升 OOD reward $R_{k'}$ 而不显著损害 $R_{\mathrm{in}(s)}$ 时：
\begin{itemize}[leftmargin=1.5em]
\item $\mathbf{e}_{i,s}$ 的 OOD 分量驱动 $\ell_{k',i,s}$ 增大。
\item $\lambda_{k',i,s}$ 从 $\sim 0$ 缓慢增长，引入少量 teacher $k'$ 的 velocity。
\item 若这确实提升了综合 reward $r_{\mathrm{composite}}$，则正向 advantage 强化这一趋势。
\end{itemize}

\textbf{本质}：$\gamma > 0$ 为已收敛的 solution 提供了``突破 single-teacher 局部最优''的梯度推力。
\end{proposition}

\section{$\gamma$ 的理论选择}
\label{sec:gamma_theory}
%% ============================================================

\subsection{$\gamma$ 过大的风险}

\begin{proposition}[$\gamma$ Upper Bound]
\label{prop:gamma_upper}
若 $\gamma$ 过大（OOD reward 占主导），则最优 $\lambda^*_{:,:,s}$ 将偏离 in-domain teacher，导致 in-domain performance 退化。具体地，当：
\begin{equation}
\gamma > \frac{\Delta R_{\mathrm{in}}}{\Delta R_{\mathrm{ood}}}
\end{equation}
其中 $\Delta R_{\mathrm{in}}$ = 从 teacher $s$ 偏离时 $R_s$ 的下降量，$\Delta R_{\mathrm{ood}}$ = 同时 OOD reward 的提升量，则 in-domain 会退化。

\textbf{经验估计}：从实验数据看，OPD-Average（完全均匀混合）导致 in-domain 下降 $\sim 0.1$，OOD 提升 $\sim 0.01$--$0.03$。因此 $\Delta R_{\mathrm{in}}/\Delta R_{\mathrm{ood}} \sim 3$--$10$。安全的 $\gamma$ 应远小于此。
\end{proposition}

\subsection{推荐 $\gamma$ 范围}

\begin{tcolorbox}[colback=blue!5, colframe=blue!40, title=推荐配置]
\begin{center}
\begin{tabular}{lcc}
\toprule
\textbf{策略} & $\gamma$ & \textbf{预期效果} \\
\midrule
保守（Level~1 为主） & $0$ & 确保 $\geq$ single teacher，无 OOD bonus \\
推荐（Level~1 + Level~2 探索） & $0.2$ & in-domain 安全，OOD 有探索空间 \\
激进（Level~2 为主） & $0.5$ & 更强 OOD 探索，in-domain 可能微降 \\
\bottomrule
\end{tabular}
\end{center}

\textbf{建议}：先用 $\gamma = 0$ 验证 Level~1 保证（确认 MoF 至少达到 single teacher），再切换 $\gamma = 0.2$ 探索 Level~2。
\end{tcolorbox}

\section{实现细节}
\label{sec:implementation}
%% ============================================================

\subsection{Prompt Set Classification}

\begin{strategy}[Prompt Set Determination]
\label{strat:classify}
每个 prompt 携带 metadata 标注其 source set。具体来说，dataloader 中每个 batch element 包含 field \texttt{prompt\_set\_id} $\in \{0, 1, \ldots, S-1\}$。

\textbf{实现}：在 \texttt{MoFTrainer.optimize()} 中：
\begin{enumerate}[leftmargin=1.5em]
\item 从 batch 中提取 \texttt{prompt\_set\_id} tensor (shape $(B,)$)
\item 对每个 set $s$：$\mathcal{B}_s = \{b : \texttt{prompt\_set\_id}[b] = s\}$
\item 使用 $\ell_{:,:,s}$ 对 $\mathcal{B}_s$ 中的 sample 计算 loss 和梯度
\end{enumerate}
\end{strategy}

\subsection{Logits 初始化}

\begin{strategy}[Initialization for Per-Set Logits]
\label{strat:init}
两种初始化策略：

\textbf{Option A: Zeros}（推荐用于验证 Level~1）
\begin{equation}
\ell_{k,i,s} = 0, \quad \forall k, i, s
\end{equation}
初始权重为均匀 $\lambda_{k,i,s} = 1/K$。所有 teacher 起点平等。

\textbf{Option B: Teacher-Biased}（推荐用于 Level~2 探索）
\begin{equation}
\ell_{k,i,s} = \begin{cases}
C & \text{if } k = s \\
0 & \text{otherwise}
\end{cases}, \quad C \approx 2.0
\end{equation}
初始权重偏向 in-domain teacher：$\lambda_{s,i,s} \approx 0.78$（$K=3$, $C=2$），其他 teacher $\approx 0.11$。从 single teacher 出发，通过 reward-driven optimization 逐步引入其他 teacher 的贡献。
\end{strategy}

\begin{remark}[Option B 的动机]
Option B 的初始点即为``near single-teacher'' solution，确保训练初期不退化。$\gamma > 0$ 的 OOD bonus 从此基础上驱动探索。若 teacher $k' \neq s$ 在某些 timestep 上有正贡献，gradient 会逐步提升 $\ell_{k',i,s}$。
\end{remark}

\subsection{EMA 与 Off-Policy}

\begin{strategy}[Per-Set EMA]
对整个 logits tensor $\ell \in \R^{K \times T \times S}$ 维护 EMA：
\begin{equation}
\bar{\ell} \leftarrow (1 - \alpha_{\mathrm{ema}})\,\bar{\ell} + \alpha_{\mathrm{ema}}\,\ell
\end{equation}
其中 $\alpha_{\mathrm{ema}} = 1 - \texttt{logits\_ema\_decay}$（如 decay $= 0.99$ → $\alpha = 0.01$）。

Sampling 使用 $\bar{\ell}$（off-policy），training 使用 $\ell$（on-policy），产生 NFT 的 new/old policy gap。
\end{strategy}

\subsection{Training Batch 组织}

\begin{strategy}[Mixed-Set Batch]
\label{strat:batch}
每个 training batch 包含来自\textbf{所有} $S$ 个 prompt set 的 sample（按比例混合）：
\begin{equation}
B = B_{\mathrm{gen}} + B_{\mathrm{pick}} + B_{\mathrm{ocr}}, \quad B_s \approx B / S
\end{equation}

\textbf{Batch 内并行}：一次 forward pass 产生所有 sample 的 teacher velocities，然后按 \texttt{prompt\_set\_id} 分组计算各 set 的 loss 和梯度。

\textbf{梯度累积}：由于三个 set 的梯度作用于 $\ell$ 的不同 slice（$\ell_{:,:,0}$, $\ell_{:,:,1}$, $\ell_{:,:,2}$），可以直接 in-place 累积，无冲突。
\end{strategy}

\subsection{Optimizer 配置}

\begin{strategy}[Per-Set Optimizer]
虽然参数解耦，但实践中使用\textbf{单个 AdamW optimizer} 管理整个 $\ell \in \R^{K \times T \times S}$（因为 Adam 的 momentum/variance 是 per-parameter 的，天然支持不同 slice 的不同更新频率）。

\begin{itemize}[leftmargin=1.5em]
\item \textbf{Learning rate}：推荐 $\texttt{logits\_lr} \in [10^{-2}, 10^{-1}]$（logits 空间需要较大 lr）。
\item \textbf{Weight decay}：$0$（logits 不需要正则化，softmax 自带 bounded output）。
\item \textbf{Gradient clipping}：$\texttt{max\_grad\_norm} = 1.0$（防止单个高 advantage sample 导致 logits 跳变过大）。
\end{itemize}
\end{strategy}

\subsection{Sampling 阶段}

\begin{algorithm}[H]
\caption{Per-Set MoF Sampling}
\label{alg:sampling}
\begin{algorithmic}[1]
\Require EMA logits $\bar{\ell}$, dataloader (mixed prompt sets)
\Ensure Samples with rewards and prompt set labels
\For{each batch from dataloader}
    \For{each sample $b$ in batch}
        \State $s_b \gets \texttt{prompt\_set\_id}[b]$
        \State Generate $x_b$ using $\bar{\ell}_{:,:,s_b}$ (Algorithm~\ref{alg:inference})
    \EndFor

\EndFor
\end{algorithmic}
\end{algorithm}

\subsection{Reward Evaluation 开销}

\begin{remark}[三倍 Reward 计算]
每个 sample 需要计算\textbf{所有三个} reward function（不仅是自己 set 的），因为 OOD bonus 需要跨 set reward。对于仅 inference-based rewards（如 PickScore, GenEval），这是轻量级的 forward pass。OCR reward 需要 OCR model inference。总 reward 评估开销 $\approx 3\times$ 单 reward 成本。
\end{remark}

\subsection{Multi-Teacher / Multi-Reward / Multi-Dataset 对应关系}

\begin{center}
\renewcommand{\arraystretch}{1.4}
\begin{tabular}{@{}lccc@{}}
\toprule
\textbf{Source (dataset)} & \textbf{In-domain Teacher} & \textbf{In-domain Reward} & \textbf{Applicable Rewards} \\
\midrule
\texttt{geneval} & GenEval Teacher & \texttt{geneval} & geneval, pick\_score \\
\texttt{pickscore} & PickScore Teacher & \texttt{pick\_score} & pick\_score \\
\texttt{ocr} & OCR Teacher & \texttt{ocr} & ocr, pick\_score \\
\bottomrule
\end{tabular}
\end{center}

\begin{definition}[对应关系的三层结构]
\begin{enumerate}[leftmargin=1.5em]
\item \textbf{Teacher $\to$ Source}：由 \texttt{TeacherConfig.sources} 指定。Teacher $k$ 的 velocity 用于所有 source 的 sample（MoF 中所有 teacher 都 forward），但 \texttt{sources} 决定了哪个 teacher 是该 source 的``in-domain teacher''（用于 \texttt{teacher\_biased} 初始化）。

\item \textbf{Source $\to$ In-domain Reward}：由 \texttt{TeacherConfig.reward\_name} 显式指定。决定了对 source $s$ 的 sample 计算 composite reward 时，哪个 reward 是主信号（权重 = 1）。

\item \textbf{Reward $\to$ Applicable Sources}：由 \texttt{RewardConfig.applicable\_sources} 指定。决定了哪些 source 的 sample 可以被该 reward model 评分（不适用的 sample 得到 NaN）。
\end{enumerate}
\end{definition}

\subsection{Reward Aggregation 公式（旧方案 vs.\ 正确方案）}

\paragraph{旧方案（raw reward 空间聚合 → 再 normalize）：}
\begin{equation}
r_{\mathrm{composite}}(x, s) = R_{\mathrm{in}(s)}(x) + \gamma \cdot \mathrm{mean}(R_{\mathrm{ood}})
\quad\xrightarrow{\text{group norm}}\quad
\mathcal{A}_i = \frac{r_i - \mu_g}{\sigma_g \cdot \sigma_{\mathrm{global}}}
\tag{旧}
\end{equation}
\textbf{问题}：GenEval $\in [0,1]$ 与 PickScore $\in [0.8, 0.95]$ 的 scale 差异使 $\gamma$ 的有效强度不可控。

\paragraph{正确方案（per-reward normalize → advantage 空间聚合）：}

与 Section~4.2 的 Definition~\ref{def:advantage} 一致：
\begin{align}
&\text{Step 1:}\quad a_k(x_i) = \frac{R_k(x_i) - \mu_{k,g}}{\sigma_{k,g}} \quad\text{(per-reward, per-group)} \tag{eq.~\ref{eq:per_reward_norm}} \\
&\text{Step 2:}\quad \mathcal{A}_{\mathrm{combined}}(x_i) = a_{\mathrm{in}(s)}(x_i) + \gamma \cdot \mathrm{mean}(a_{\mathrm{ood}}) \tag{eq.~\ref{eq:advantage_aggregation}} \\
&\text{Step 3:}\quad \mathcal{A}_{\mathrm{final}}(x_i) = \mathcal{A}_{\mathrm{combined}}(x_i) / \sigma_{\mathrm{global}} \tag{eq.~\ref{eq:final_norm}}
\end{align}

此方案中 $\gamma = 0.2$ 的语义清晰：OOD advantage（已归一化为 $\sim N(0,1)$）贡献 20\% 的信号强度，与 reward 原始 scale 完全无关。

\subsection{具体示例（3-teacher config）}

\begin{itemize}[leftmargin=1.5em]
\item \texttt{geneval} sample: $\mathcal{A} = a_{\mathrm{geneval}} + 0.2 \cdot a_{\mathrm{pick\_score}}$
\item \texttt{pickscore} sample: $\mathcal{A} = a_{\mathrm{pick\_score}}$（无 OOD applicable）
\item \texttt{ocr} sample: $\mathcal{A} = a_{\mathrm{ocr}} + 0.2 \cdot a_{\mathrm{pick\_score}}$
\end{itemize}

其中每个 $a_k$ 已经是 group-normalized（zero-mean, unit-variance within group）。

\subsection{分布式训练下的正确性要求}

\begin{tcolorbox}[colback=red!5, colframe=red!50, title=关键约束]
在 \texttt{distributed\_k\_repeat} sampler 下，同一 group 的 $K$ 个 sample 可能分散在不同 rank 上。正确的 group-mean 需要\textbf{跨 rank gather}——只用本地 samples 计算 group-mean 会得到错误结果。

当前实现通过 \texttt{AdvantageProcessor.collect\_group\_rewards()} 解决：
\begin{enumerate}[leftmargin=1.5em]
\item Pack rewards + unique\_id → \texttt{(B, 2)} tensor
\item \texttt{accelerator.gather()} → \texttt{(W*B, 2)} global tensor
\item 在 global tensor 上计算 group-mean / global-std
\item \texttt{\_to\_local()} scatter 回各 rank
\end{enumerate}
\end{tcolorbox}

\subsection{实现中的潜在问题分析}

\begin{enumerate}[leftmargin=1.5em]

\item \textbf{Cross-source Group 混合问题}（中等风险）

当前 sampling 使用 \texttt{\_interleaved\_source\_iter()} 交替产出不同 source 的 batch。每个 batch 对应一个 source，batch 内的 samples 共享相同的 \texttt{unique\_id}（同 prompt 的 $K$ 个 sample 在同一 batch 内）。

\textbf{但}：如果 \texttt{group\_size=16} 且 \texttt{per\_device\_batch\_size=8}，则一个 group 需要 2 个 batch 来生成 16 个 sample（batch size $\times$ 8 GPU $= 64$ per batch，但 unique\_sample\_num\_per\_epoch 控制不同 prompt 的数量）。

\textbf{关键问题}：同一 group 内的所有 sample 必须来自同一 source（因为它们共享同一个 prompt）。当前的 \texttt{\_interleaved\_source\_iter()} 保证每个 batch 是单 source 的，但同一 prompt 的 $K$ 个重复 sample 可能跨多个 batch。由于 interleaved 模式 round-robin 切换 source，这些重复 sample 确实在同一个 source 的 batch 中生成——\textbf{正确，无问题}。

\item \textbf{Group-Std vs.\ Global-Std 的语义差异}（低风险但需注意）

当前实现做了两层 normalization：先 group-std（eq.~\ref{eq:per_reward_norm}），再 global-std（eq.~\ref{eq:final_norm}）。这等价于：
\begin{equation}
\mathcal{A}_i = \frac{r_i - \mu_g}{\sigma_g \cdot \sigma_{\mathrm{global}}}
\end{equation}

如果不同 source 的 group 内 reward variance 差异很大（如 geneval 的 group-std $\approx 0.3$，pickscore 的 group-std $\approx 0.02$），group-std normalization 后它们的 scale 已经对齐。global-std 此时主要作用是稳定整体 scale（不改变相对大小）。

\textbf{替代方案}：只用 group-mean 减去（不除 group-std），然后用 global-std 统一 normalize。这保留了不同 source 的 reward 信号强度差异，可能更好：
\begin{equation}
\mathcal{A}_i^{\mathrm{alt}} = \frac{r_i - \mu_g}{\sigma_{\mathrm{global}}}
\end{equation}

但当前的双层 normalization 在 GDPO 论文中已验证有效，暂不修改。

\item \textbf{Composite Reward 的 Scale 不一致}（需注意）

由于 \texttt{applicable\_sources} 过滤，不同 source 的 sample 的 composite reward 组成不同：
\begin{itemize}
\item geneval sample: $R_{\mathrm{geneval}} + 0.2 \cdot R_{\mathrm{pick}}$（两项之和）
\item pickscore sample: $R_{\mathrm{pick}}$（单项）
\end{itemize}

两者的原始 scale 不同。但由于 group-mean 是\textbf{组内}计算（同一 prompt $\Rightarrow$ 同一 source $\Rightarrow$ 同一 composite 公式），且 group-std 归一化后 scale 对齐，这\textbf{不是问题}。

\item \textbf{\texttt{\_to\_local()} 返回类型}（minor bug）

\texttt{AdvantageProcessor.\_to\_local()} 在 \texttt{group\_on\_same\_rank=True} 时返回 \texttt{torch.Tensor}（可能在 GPU 上），在 \texttt{group\_on\_same\_rank=False} 时返回 CPU numpy array 经 \texttt{torch.as\_tensor} 转换。后续 \texttt{float(adv)} 对 GPU tensor 有隐式 D2H transfer。建议统一 \texttt{.cpu().numpy()} 后再 store。

\end{enumerate}

\subsection{推荐的实现改进}

\begin{tcolorbox}[colback=green!5, colframe=green!50!black, title=短期可执行的改进]
\begin{enumerate}[leftmargin=1.5em]
\item \textbf{fix}: 在 \texttt{prepare\_feedback} 中 \texttt{local\_advantages} 转为 numpy 后再逐 sample 存储（避免逐个 GPU→CPU transfer）。

\item \textbf{enhancement}: 为 pickscore source 也配置 OOD reward（如 geneval），使其也能受益于 cross-teacher signal：
\begin{verbatim}
- name: "geneval"
  applicable_sources: [geneval, pickscore]
\end{verbatim}
这样 pickscore sample 的 composite = $R_{\mathrm{pick}} + 0.2 \cdot R_{\mathrm{geneval}}$，获得更丰富的梯度信号。

\item \textbf{monitoring}: 添加 per-source advantage mean 的 logging，确认各 source 的 advantage 分布是否平衡（不应有某个 source 的 advantage 系统性偏高或偏低）。
\end{enumerate}
\end{tcolorbox}

%% ============================================================
\section{Batch-Level Normalization 时机分析}
\label{sec:norm_timing}
%% ============================================================

核心问题：在 MoF 的 3-step advantage pipeline 中，batch-level normalization（$\sigma_{\mathrm{global}}$ 除法）应放在\textbf{per-reward advantage 聚合之前}还是\textbf{之后}？

\subsection{两种方案定义}

\begin{definition}[Option A: Batch-Level Normalization \textbf{Before} Aggregation]
\label{def:norm_before}
\begin{align}
&\text{Step 1:}\quad a_k(x_i) = \frac{R_k(x_i) - \mu_{k,g}}{\max(\sigma_{k,g}, \epsilon)} \quad\text{(per-reward, per-group)} \\
&\text{Step 1.5:}\quad \hat{a}_k(x_i) = \frac{a_k(x_i)}{\max(\sigma_{k,\mathrm{batch}}, \epsilon)}, \quad \sigma_{k,\mathrm{batch}} = \mathrm{std}(\{a_k(x_j)\}_{j=1}^N) \\
&\text{Step 2:}\quad \mathcal{A}_{\mathrm{final}}(x_i) = \hat{a}_{\mathrm{in}(s)}(x_i) + \gamma \cdot \frac{1}{|\mathcal{R}_{\mathrm{ood}}|}\sum_{j \in \mathcal{R}_{\mathrm{ood}}} \hat{a}_j(x_i)
\end{align}
特点：每个 reward 的 advantage 被独立归一化为 batch-level unit variance，然后直接聚合。\textbf{聚合后不再做额外 normalization}。
\end{definition}

\begin{definition}[Option B: Batch-Level Normalization \textbf{After} Aggregation]
\label{def:norm_after}
\begin{align}
&\text{Step 1:}\quad a_k(x_i) = \frac{R_k(x_i) - \mu_{k,g}}{\max(\sigma_{k,g}, \epsilon)} \quad\text{(per-reward, per-group)} \\
&\text{Step 2:}\quad \mathcal{A}_{\mathrm{combined}}(x_i) = a_{\mathrm{in}(s)}(x_i) + \gamma \cdot \frac{1}{|\mathcal{R}_{\mathrm{ood}}|}\sum_{j \in \mathcal{R}_{\mathrm{ood}}} a_j(x_i) \\
&\text{Step 3:}\quad \mathcal{A}_{\mathrm{final}}(x_i) = \frac{\mathcal{A}_{\mathrm{combined}}(x_i)}{\max(\sigma_{\mathrm{global}}, \epsilon)}, \quad \sigma_{\mathrm{global}} = \mathrm{std}(\{\mathcal{A}_{\mathrm{combined}}(x_j)\}_{j=1}^N)
\end{align}
特点：先在 advantage 空间做 in-domain + OOD 聚合，然后对聚合后的 combined advantage 做统一的 batch-level normalization。
\end{definition}

\subsection{数学等价性分析}

\begin{proposition}[Group-Std 后 Batch-Std 近似为恒等]
\label{prop:batch_std_identity}
设 Step 1 的 group normalization 正确工作（即各 group 内 sample 数量 $\geq 4$），则：
\begin{equation}
\sigma_{k,\mathrm{batch}} = \mathrm{std}\left(\left\{\frac{R_k(x_j) - \mu_{k,g_j}}{\sigma_{k,g_j}}\right\}_{j=1}^N\right) \approx 1
\end{equation}
\textbf{原因}：各 group 独立贡献 zero-mean, unit-variance 的 $a_k$ 值。跨 group 拼接后的 batch-level std 仍为 $\approx 1$（除非 group 数量极少或 group 间 reward 分布高度非平稳）。

因此 Option A 的 Step 1.5 近似为恒等操作：$\hat{a}_k \approx a_k$。
\end{proposition}

\begin{corollary}[两种方案的差异来源]
\label{cor:diff_source}
Option A 与 Option B 的实质差异不在于``是否消除 per-reward scale 差异''（Step 1 的 group-std 已经做了），而在于：
\begin{itemize}[leftmargin=1.5em]
\item Option A：\textbf{聚合后不再 normalize}。$\mathcal{A}_{\mathrm{final}}$ 的 variance 取决于 source $s$ 的 OOD reward 数量：
\begin{align}
\mathrm{Var}(\mathcal{A}^A_{\mathrm{final}}) &= 1 + \frac{\gamma^2}{|\mathcal{R}_{\mathrm{ood}}|} \quad\text{（假设各 $a_k$ 独立）}
\end{align}
对 geneval ($|\mathcal{R}_{\mathrm{ood}}| = 1$): $\mathrm{Var} = 1 + 0.04 = 1.04$\\
对 pickscore ($|\mathcal{R}_{\mathrm{ood}}| = 0$): $\mathrm{Var} = 1$\\
对 ocr ($|\mathcal{R}_{\mathrm{ood}}| = 1$): $\mathrm{Var} = 1.04$

$\Rightarrow$ 不同 source 的 final advantage 有微小的 variance 差异。

\item Option B：$\sigma_{\mathrm{global}}$ 统一 rescale 所有 sample，确保 $\mathrm{Var}(\mathcal{A}^B_{\mathrm{final}}) \approx 1$ 对所有 sample。
\end{itemize}
\end{corollary}

\subsection{对 NFT Loss 的影响}

\begin{proposition}[NFT Loss 对 Advantage Scale 的敏感性]
\label{prop:nft_sensitivity}
回顾 NFT loss 中 advantage 的作用（eq.~\ref{eq:normalized_adv}）：
\begin{equation}
r_s = \frac{\adv_s + A_{\max}}{2A_{\max}} \in [0, 1]
\end{equation}
其中 $A_{\max} = \max(|\mathrm{adv\_clip\_range}|)$（配置中为 5.0）。

\begin{itemize}[leftmargin=1.5em]
\item $A_{\max}$ 起到 normalization 的作用：将 advantage 映射到 $[0,1]$ 区间。
\item 若 $\mathcal{A}_{\mathrm{final}}$ 的实际 std $\ll A_{\max}$（如 std $\approx 1$ vs $A_{\max} = 5$），则 $r_s \approx 0.5 \pm 0.1$——advantage 的有效信号强度较弱。
\item 若 $\mathcal{A}_{\mathrm{final}}$ 的 std 变动较大，则不同 training epoch 的 $r_s$ 分布不稳定。
\end{itemize}

\textbf{结论}：NFT loss 对 advantage 的 absolute scale 敏感度较低（通过 $A_{\max}$ 缓冲），但对\textbf{不同 sample 间的相对大小}高度敏感。因此 Option A 和 B 在 NFT loss 下的行为差异很小（$\gamma = 0.2$ 导致的 variance 差异仅 4\%）。
\end{proposition}

\subsection{推荐方案与理论依据}

\begin{tcolorbox}[colback=blue!5, colframe=blue!50, title=推荐：Option B（After Aggregation）]
\textbf{推荐 batch-level normalization 放在 advantage 聚合\textbf{之后}}，理由如下：

\begin{enumerate}[leftmargin=1.5em]
\item \textbf{Scale 消除已由 Step 1 完成}：Per-reward group-std 已将各 reward 归一化到 $\sim N(0,1)$。Pre-aggregation batch-std 近似为恒等操作（Proposition~\ref{prop:batch_std_identity}），增加计算但无实质收益。

\item \textbf{Post-aggregation $\sigma_{\mathrm{global}}$ 提供 robustness}：当 group size 较小（如 $K = 4$）时，per-group std 估计可能不精确，导致聚合后的 advantage 的 batch-level std 偏离 1。$\sigma_{\mathrm{global}}$ 除法提供了一个``safety net''，确保最终 advantage 的 scale 稳定。

\item \textbf{统一 scale 对 NFT loss 友好}：虽然 NFT loss 对 absolute scale 不太敏感（有 $A_{\max}$ 缓冲），但统一的 advantage scale 使 learning rate 的选择更稳定——不需要根据 OOD reward 数量调整 lr。

\item \textbf{与 GRPO/GDPO 一致}：标准 GRPO 使用 ``group-mean → global-std'' 的两层结构，MoF 的 ``per-reward group-norm → aggregate → global-std'' 是其自然扩展。
\end{enumerate}
\end{tcolorbox}

\begin{remark}[何时 Option A 可能更好]
若未来引入 reward 间的\textbf{非独立性}（如 geneval reward 与 pickscore reward 高度相关），则 $\mathrm{Var}(\mathcal{A}_{\mathrm{combined}})$ 可能显著偏离 $1 + \gamma^2/|\mathcal{R}_{\mathrm{ood}}|$ 的独立假设。此时 pre-aggregation normalization 可以消除相关性导致的 variance inflation。但在当前配置下（rewards 相对独立），Option B 更简洁有效。
\end{remark}

\subsection{最终 Pipeline 伪代码}

\begin{algorithm}[H]
\caption{MoF Advantage Pipeline (Recommended: Option B)}
\label{alg:advantage_pipeline}
\begin{algorithmic}[1]
\Require Raw rewards $\{R_k(x_i)\}_{k,i}$, group indices $\{g_i\}$, source labels $\{s_i\}$, gamma $\gamma$
\Ensure Per-sample advantage $\{\mathcal{A}_i\}$
\Statex
\Statex \textbf{// Step 1: Per-Reward Group Normalization}
\For{each reward $k$}
    \For{each group $g$}
        \State $\mu_{k,g} \gets \mathrm{mean}(\{R_k(x_j) : g_j = g, R_k(x_j) \neq \mathrm{NaN}\})$
        \State $\sigma_{k,g} \gets \mathrm{std}(\{R_k(x_j) : g_j = g, R_k(x_j) \neq \mathrm{NaN}\})$
        \For{$x_j$ in group $g$ where $R_k(x_j) \neq \mathrm{NaN}$}
            \State $a_k(x_j) \gets (R_k(x_j) - \mu_{k,g}) / \max(\sigma_{k,g}, \epsilon)$
        \EndFor
    \EndFor
\EndFor
\Statex
\Statex \textbf{// Step 2: Per-Set Advantage Aggregation}
\For{each sample $x_i$ from source $s$}
    \State $\mathrm{in} \gets$ in-domain reward for source $s$
    \State $\mathcal{R}_{\mathrm{ood}} \gets$ applicable non-in-domain rewards for $x_i$ (non-NaN)
    \State $\mathcal{A}_{\mathrm{combined}}(x_i) \gets a_{\mathrm{in}}(x_i) + \gamma \cdot \mathrm{mean}(\{a_j(x_i)\}_{j \in \mathcal{R}_{\mathrm{ood}}})$
\EndFor
\Statex
\Statex \textbf{// Step 3: Global-Std Normalization}
\State $\sigma_{\mathrm{global}} \gets \mathrm{std}(\{\mathcal{A}_{\mathrm{combined}}(x_j)\}_{j=1}^N)$
\For{each sample $x_i$}
    \State $\mathcal{A}_{\mathrm{final}}(x_i) \gets \mathcal{A}_{\mathrm{combined}}(x_i) / \max(\sigma_{\mathrm{global}}, \epsilon)$
    \State $\mathcal{A}_{\mathrm{final}}(x_i) \gets \mathrm{clip}(\mathcal{A}_{\mathrm{final}}(x_i),\, \mathrm{adv\_clip\_min},\, \mathrm{adv\_clip\_max})$
\EndFor
\end{algorithmic}
\end{algorithm}

\section{梯度消失分析与解决方案}
\label{sec:gradient_vanishing}
%% ============================================================

\subsection{问题现象}

实验观察到 MoF 训练中 \texttt{grad\_norm} $\sim 10^{-10}$，且：
\begin{itemize}[leftmargin=1.5em]
\item 修改 \texttt{nft\_beta} = 1.0, 0.1, 0.01 只改变 loss 的 scale，不改变 grad\_norm 的数量级。
\item Teacher-biased 初始化（$C=2$）和 zeros 初始化表现相同数量级。
\item EMA 与 current logits 的关系对 grad\_norm 无显著影响。
\end{itemize}

\subsection{根因分析：逐步推导}

我们从 loss $\loss$ 到 logits $\ell_{j,i,s}$ 的完整梯度链出发，逐步展示每个环节如何引入梯度压缩。

\subsubsection{Step 1: 明确计算图}

MoF 的前向计算图（单 timestep $i$, 单 set $s$, 省略下标 $i,s$）：
\begin{equation}
\underbrace{\ell \in \R^K}_{\text{logits}} \xrightarrow{\text{softmax}} \underbrace{\lambda \in \Delta^K}_{\text{weights}} \xrightarrow{\text{linear mix}} \underbrace{v^{\mathrm{new}} = \sum_k \lambda_k v^{(k)} \in \R^d}_{\text{combined velocity}} \xrightarrow{\text{NFT loss}} \loss \in \R
\end{equation}
其中 $v^{(k)} \in \R^d$ 是 teacher $k$ 的 \textbf{detached}（常数）velocity，$d = C \times H \times W = 65536$。

\subsubsection{Step 2: Loss $\to$ Combined Velocity 的梯度 $\mathbf{g}$}

定义 loss 对 combined velocity 的梯度向量：
\begin{equation}
\mathbf{g} \triangleq \frac{\partial \loss}{\partial v^{\mathrm{new}}} \in \R^d
\end{equation}

这是一个 $d$-维向量，由 NFT 的 MSE loss 决定。其具体形式为（展开 NFT loss）：

NFT loss 对 $v^{\mathrm{new}}$ 的梯度来自：
\begin{align}
\loss &= \frac{1}{\beta}\left[r \cdot L^+(v^+) + (1-r) \cdot L^-(v^-)\right] \\
v^+ &= \beta\, v^{\mathrm{new}} + (1-\beta)\, v^{\mathrm{old}} \;\Rightarrow\; \frac{\partial v^+}{\partial v^{\mathrm{new}}} = \beta\, I_d \\
v^- &= (1+\beta)\, v^{\mathrm{old}} - \beta\, v^{\mathrm{new}} \;\Rightarrow\; \frac{\partial v^-}{\partial v^{\mathrm{new}}} = -\beta\, I_d
\end{align}
其中 $v^{\mathrm{old}}$ 是 detached 的 EMA combined velocity。代入链式法则：
\begin{align}
\mathbf{g} = \frac{\partial \loss}{\partial v^{\mathrm{new}}} &= \frac{1}{\beta}\left[r \cdot \frac{\partial L^+}{\partial v^+} \cdot \beta + (1-r) \cdot \frac{\partial L^-}{\partial v^-} \cdot (-\beta)\right] \notag \\
&= r \cdot \frac{\partial L^+}{\partial v^+} - (1-r) \cdot \frac{\partial L^-}{\partial v^-}
\label{eq:g_beta_cancels}
\end{align}

\begin{tcolorbox}[colback=yellow!5, colframe=yellow!60!black, title=关键发现 1: $\beta$ 消去]
$\beta$ 在梯度中完全消去！这数学性地解释了为什么修改 \texttt{nft\_beta} = 1.0, 0.1, 0.01 不影响 grad\_norm——只影响 loss 的绝对值（scaling），不影响 $\partial\loss/\partial v^{\mathrm{new}}$ 的方向或大小。
\end{tcolorbox}

$\mathbf{g}$ 的每个分量 $\mathbf{g}[c,h,w]$ 由 NFT 的 self-normalized MSE 给出：
\begin{equation}
\mathbf{g}[c,h,w] = r \cdot \frac{2(x^+_0 - x_0)[c,h,w] \cdot (-\sigma_i)}{(w^+)^2 \cdot d} - (1-r) \cdot \frac{2(x^-_0 - x_0)[c,h,w] \cdot \sigma_i}{(w^-)^2 \cdot d}
\end{equation}
量级：每个分量 $|\mathbf{g}[c,h,w]| \sim \frac{2\sigma_i \cdot |2r-1|}{d} \sim \frac{2 \times 0.45 \times 0.2}{65536} \approx 2.7 \times 10^{-6}$。\\
向量范数：$\|\mathbf{g}\| = \sqrt{\sum_{c,h,w} \mathbf{g}[c,h,w]^2} \sim \sqrt{d} \cdot 2.7 \times 10^{-6} \approx 6.9 \times 10^{-4}$。

\subsubsection{Step 3: Combined Velocity $\to$ Weights 的梯度}

由 $v^{\mathrm{new}} = \sum_k \lambda_k\, v^{(k)}$（$v^{(k)}$ 是常数），loss 对 weight $\lambda_k$ 的梯度为标量：
\begin{equation}
\frac{\partial \loss}{\partial \lambda_k} = \mathbf{g}^\top v^{(k)} = \sum_{c,h,w} \mathbf{g}[c,h,w] \cdot v^{(k)}[c,h,w]
\label{eq:grad_wrt_lambda}
\end{equation}

这是两个 $d$-维向量的 \textbf{内积}（dot product），结果是标量。

\textbf{量级}：$\mathbf{g}$ 的每个分量 $\sim 10^{-6}$，$v^{(k)}$ 的每个分量 $\sim O(1)$。若两者存在非零相关性（alignment $\sim 30\%$），则：
\begin{equation}
\left|\frac{\partial \loss}{\partial \lambda_k}\right| \approx d \cdot 2.7 \times 10^{-6} \cdot 1.0 \cdot 0.3 \approx 0.05
\end{equation}
\textbf{注意：此时梯度量级完全合理（$\sim 0.05$）。}

\subsubsection{Step 4: Weights $\to$ Logits 的梯度（Softmax Jacobian — 灾难发生处）}

Softmax 的 Jacobian 矩阵为：
\begin{equation}
\frac{\partial \lambda_k}{\partial \ell_j} = \frac{1}{\tau}\,\lambda_k\left(\delta_{kj} - \lambda_j\right)
\end{equation}
其中：
\begin{itemize}[leftmargin=1.5em]
\item $\lambda_k = \softmax(\ell / \tau)_k$ 是 teacher $k$ 的当前 mixing weight（softmax 输出的第 $k$ 个分量）
\item $\ell_j$ 是 teacher $j$ 对应的 logit 参数（我们要求梯度的目标参数）
\item $\delta_{kj}$ 是 Kronecker delta：$\delta_{kj} = 1$ if $k = j$, else $0$
\item $\tau$ 是 softmax temperature
\end{itemize}

通过链式法则，loss 对 logit $\ell_j$ 的梯度为：
\begin{align}
\frac{\partial \loss}{\partial \ell_j} &= \sum_{k=1}^K \frac{\partial \loss}{\partial \lambda_k} \cdot \frac{\partial \lambda_k}{\partial \ell_j} \\
&= \sum_k \underbrace{\left(\mathbf{g}^\top v^{(k)}\right)}_{\text{Step 3 的结果}} \cdot \underbrace{\frac{\lambda_k(\delta_{kj} - \lambda_j)}{\tau}}_{\text{softmax Jacobian}}
\label{eq:step4_chain}
\end{align}
其中 $\mathbf{g}^\top v^{(k)}$ 是 Step 3 中得到的 loss 对 $\lambda_k$ 的梯度（标量），表示``如果增加 teacher $k$ 的权重，loss 会如何变化''。

展开求和（将 $k = j$ 和 $k \neq j$ 的项分别处理）：
\begin{align}
\frac{\partial \loss}{\partial \ell_j} &= \frac{1}{\tau}\sum_k \left(\mathbf{g}^\top v^{(k)}\right) \lambda_k(\delta_{kj} - \lambda_j) \notag \\
&= \frac{1}{\tau}\left[\underbrace{\left(\mathbf{g}^\top v^{(j)}\right) \lambda_j(1 - \lambda_j)}_{k = j \text{ 的贡献}} + \underbrace{\sum_{k \neq j}\left(\mathbf{g}^\top v^{(k)}\right) \lambda_k(0 - \lambda_j)}_{k \neq j \text{ 的贡献}}\right] \notag \\
&= \frac{\lambda_j}{\tau}\left[\left(\mathbf{g}^\top v^{(j)}\right)(1 - \lambda_j) - \sum_{k \neq j} \lambda_k \left(\mathbf{g}^\top v^{(k)}\right)\right] \notag \\
&= \frac{\lambda_j}{\tau}\left[\mathbf{g}^\top v^{(j)} - \lambda_j\,\mathbf{g}^\top v^{(j)} - \sum_{k \neq j} \lambda_k\, \mathbf{g}^\top v^{(k)}\right] \notag \\
&= \frac{\lambda_j}{\tau}\left[\mathbf{g}^\top v^{(j)} - \sum_{k=1}^K \lambda_k\, \mathbf{g}^\top v^{(k)}\right]
\label{eq:softmax_detailed}
\end{align}
其中最后一步利用了 $\lambda_j\,\mathbf{g}^\top v^{(j)} + \sum_{k \neq j} \lambda_k\, \mathbf{g}^\top v^{(k)} = \sum_k \lambda_k\, \mathbf{g}^\top v^{(k)}$。

定义加权均值 velocity $\bar{v} = \sum_k \lambda_k v^{(k)}$（即 combined velocity $v^{\mathrm{new}}$ 本身），注意到：
\begin{equation}
\sum_k \lambda_k\, \mathbf{g}^\top v^{(k)} = \mathbf{g}^\top \left(\sum_k \lambda_k\, v^{(k)}\right) = \mathbf{g}^\top \bar{v}
\end{equation}

代入得最终结果：
\begin{equation}
\boxed{
\frac{\partial \loss}{\partial \ell_j} = \frac{\lambda_j}{\tau}\,\mathbf{g}^\top\left(v^{(j)} - \bar{v}\right)
}
\label{eq:final_gradient_derivation}
\end{equation}
其中：
\begin{itemize}[leftmargin=1.5em]
\item $\ell_j$：teacher $j$ 的 logit 参数（可学习标量）
\item $\lambda_j$：teacher $j$ 当前的 softmax 权重（$\in [0, 1]$, 由 $\ell$ 决定）
\item $\tau$：softmax temperature（超参数）
\item $\mathbf{g} = \partial\loss / \partial v^{\mathrm{new}} \in \R^d$：loss 对 combined velocity 的梯度向量（由 NFT loss 的 MSE 决定）
\item $v^{(j)} \in \R^d$：teacher $j$ 的 velocity 输出（frozen, detached 常数向量）
\item $\bar{v} = \sum_k \lambda_k v^{(k)} \in \R^d$：所有 teacher 的加权均值 velocity（= combined velocity $v^{\mathrm{new}}$）
\item $v^{(j)} - \bar{v} \in \R^d$：teacher $j$ 偏离当前混合均值的``差异方向''
\item $\mathbf{g}^\top(v^{(j)} - \bar{v})$：$d$-维向量的\textbf{内积}（标量），衡量 loss gradient 方向与 teacher 差异方向的对齐程度
\end{itemize}

\begin{tcolorbox}[colback=red!5, colframe=red!60!black, title=关键发现 2: Softmax 的``均值减除''效应]
比较 Step 3 和 Step 4：
\begin{align}
\text{对 } \lambda_j \text{ 的梯度：} &\quad \frac{\partial \loss}{\partial \lambda_j} = \mathbf{g}^\top v^{(j)} \quad\text{（量级 $\sim 0.05$）} \\
\text{对 } \ell_j \text{ 的梯度：} &\quad \frac{\partial \loss}{\partial \ell_j} = \frac{\lambda_j}{\tau}\,\mathbf{g}^\top\underbrace{\left(v^{(j)} - \bar{v}\right)}_{\text{!!!}} \quad\text{（量级 $\sim 10^{-9}$）}
\end{align}

Softmax Jacobian 将``$\mathbf{g}$ 对 $v^{(j)}$ 的 dot product''替换为``$\mathbf{g}$ 对 $(v^{(j)} - \bar{v})$ 的 dot product''。

\textbf{直觉理解}：Softmax 的归一化约束（$\sum_k \lambda_k = 1$）意味着增大 $\ell_j$ 必须同时减小其他 $\ell_k$ 的有效权重。梯度必须衡量``teacher $j$ 比\textbf{当前混合}好多少''，而非``teacher $j$ 绝对好不好''。这个\textbf{相对比较}就是 $(v^{(j)} - \bar{v})$ 项的来源。

当所有 teacher 都来自同一 base model 时：$v^{(j)} \approx \bar{v} \approx v_{\mathrm{base}}$（因为 LoRA perturbation 很小），所以 $(v^{(j)} - \bar{v})$ 的量级仅为 full velocity 的 1--5\%。这就是灾难所在。
\end{tcolorbox}

\subsubsection{Step 5: 代入 LoRA 结构量化}

对 LoRA teacher：$v^{(k)} = v_{\mathrm{base}} + \delta^{(k)}$，因此：
\begin{equation}
v^{(j)} - \bar{v} = (v_{\mathrm{base}} + \delta^{(j)}) - (v_{\mathrm{base}} + \bar{\delta}) = \delta^{(j)} - \bar{\delta}
\end{equation}

\textbf{Base model velocity 完全消去}——logit 梯度仅由 LoRA 残差的差异驱动。

对于 $K=2$ 均匀权重（$\lambda_0 = \lambda_1 = 0.5$）：
\begin{equation}
v^{(0)} - \bar{v} = \frac{1}{2}(v^{(0)} - v^{(1)}) = \frac{1}{2}(\delta^{(0)} - \delta^{(1)})
\end{equation}

最终：
\begin{equation}
\frac{\partial \loss}{\partial \ell_0} = \frac{0.5}{\tau}\,\mathbf{g}^\top \frac{\delta^{(0)} - \delta^{(1)}}{2} = \frac{1}{4\tau}\,\mathbf{g}^\top(\delta^{(0)} - \delta^{(1)})
\end{equation}

量级：$|\mathbf{g}^\top(\delta^{(0)} - \delta^{(1)})| \sim \|\mathbf{g}\| \cdot \|\delta^{(0)} - \delta^{(1)}\| \cdot |\cos\theta|$，其中：
\begin{itemize}[leftmargin=1.5em]
\item $\|\mathbf{g}\| \sim 6.9 \times 10^{-4}$
\item $\|\delta^{(0)} - \delta^{(1)}\| \sim \sqrt{d} \cdot 0.02 \approx 5.1$（LoRA diff per element $\sim 0.02$）
\item $|\cos\theta| \sim 1/\sqrt{d} \approx 0.004$（$\mathbf{g}$ 和 $\delta$-diff 方向无先验相关性）
\end{itemize}

\begin{equation}
\left|\frac{\partial \loss}{\partial \ell_0}\right| \sim \frac{1}{4} \times 6.9 \times 10^{-4} \times 5.1 \times 0.004 \approx 3.5 \times 10^{-6}
\end{equation}

再除以 GAS $= 80$：$\approx 4.4 \times 10^{-8}$。对 $K \times T \times S = 40$ 个参数的 grad norm：$\sqrt{40} \times 4.4 \times 10^{-8} \approx 2.8 \times 10^{-7}$。

考虑到 $|\cos\theta|$ 的实际值可能更小（LoRA diff 和 MSE gradient 几乎正交），进一步压缩 1--2 个数量级到 $\sim 10^{-9}$--$10^{-10}$。

\subsection{为什么 \texttt{nft\_beta} 不影响梯度}

如 Step 2 公式~\eqref{eq:g_beta_cancels} 所示，$\mathbf{g} = r \cdot \partial L^+/\partial v^+ - (1-r) \cdot \partial L^-/\partial v^-$，与 $\beta$ 完全无关。$\beta$ 只出现在 $v^+, v^-$ 的\textbf{定义}中和 loss 的 $1/\beta$ 前缀中，两者的贡献在链式法则中精确消去。

\subsection{梯度消失的三重因素总结}

\begin{tcolorbox}[colback=red!5, colframe=red!60!black, title=梯度消失的三重因素]
\begin{equation}
\underbrace{|\nabla_\ell \loss|}_{\sim 10^{-10}} \;=\;
\underbrace{\frac{\lambda_j}{\tau} \cdot |\mathbf{g}|}_{\text{NFT loss gradient}} \;\times\;
\underbrace{|\cos\theta| \cdot \|\delta^{(j)} - \bar{\delta}\|}_{\text{Softmax direction projection}} \;\times\;
\underbrace{\frac{1}{\text{GAS} \cdot \text{DDP}}}_{\text{Distributed scaling}}
\end{equation}

\begin{enumerate}[leftmargin=1.5em]
\item \textbf{Softmax Mean Subtraction}（主因，$\sim 100\times$ reduction）：

Softmax 的 Jacobian 使梯度正比于 $v^{(j)} - \bar{v}$——teacher $j$ 偏离加权均值的方向。对于共享 base model 的 LoRA teacher：
\begin{equation}
\|v^{(j)} - \bar{v}\| = \|\delta^{(j)} - \bar{\delta}\| \ll \|v^{(j)}\|
\end{equation}
LoRA perturbation 仅占 base velocity 的 1--5\%。\textbf{任何 normalized mixing（softmax, sigmoid for $K$=2, 或任何满足 $\sum w_k = 1$ 的参数化）都有此性质}。

\item \textbf{High-Dimensional Dot Product Cancellation}（$\sim 10$--$100\times$ reduction）：

梯度是 $\mathbf{g} \in \R^d$ 与 $(v^{(j)} - \bar{v}) \in \R^d$ 的 dot product。两者方向无先验相关性（loss gradient 指向``prediction error 方向''，而 teacher difference 指向``LoRA 差异方向''），在 $d = 65536$ 维空间中：
\begin{equation}
\E[|\cos\theta|] \sim \frac{1}{\sqrt{d}} \approx 0.004
\end{equation}

\item \textbf{GAS + DDP Division}（$\sim 640\times$ reduction）：

Gradient accumulation steps = $\frac{\text{num\_batches}}{\text{gradient\_step\_per\_epoch}} \times T = 8 \times 10 = 80$，加上 DDP 8-GPU all-reduce averaging。
\end{enumerate}

\textbf{总压缩比}: $\sim 0.02 \times 0.004 \times (1/80) \approx 10^{-6}$，将 $\sim 10^{-4}$ 量级的``理想梯度''压缩至 $\sim 10^{-10}$。
\end{tcolorbox}

\subsection{核心困难的数学本质}

\begin{theorem}[Normalized Mixing 的梯度瓶颈]
\label{thm:fundamental_limit}
对\textbf{任何}满足 $\sum_k w_k = 1$ 的参数化 $w_k = f_k(\theta)$（包括 softmax, sigmoid for $K$=2, Gumbel-softmax 等），combined velocity $v_\theta = \sum_k w_k v^{(k)}$ 的梯度满足：
\begin{equation}
\frac{\partial v_\theta}{\partial \theta} = \sum_k \frac{\partial w_k}{\partial \theta}\, v^{(k)} = \sum_k \frac{\partial w_k}{\partial \theta}\, (v^{(k)} - v_\theta + v_\theta)
\end{equation}
由约束 $\sum_k w_k = 1$ $\Rightarrow$ $\sum_k \frac{\partial w_k}{\partial \theta} = 0$，因此：
\begin{equation}
\boxed{
\frac{\partial v_\theta}{\partial \theta} = \sum_k \frac{\partial w_k}{\partial \theta}\, (v^{(k)} - v_\theta)
}
\label{eq:normalized_constraint}
\end{equation}

\textbf{梯度恒等式}：无论参数化方式如何，normalized mixing 的梯度\textbf{必定}正比于 teacher velocities 与其加权均值的偏差。当 teachers 共享 base model 时，此偏差本质上是 LoRA residual 差异——量级远小于 velocity 本身。

\textbf{这是数学约束，非 softmax 特有}。
\end{theorem}

\subsection{对比：为什么 LLM 的 Logits + Softmax 没有此问题}

\begin{tcolorbox}[colback=blue!5, colframe=blue!50!black, title=对比：LLM Head (Softmax) vs MoF (Softmax)]

\textbf{LLM 的设置}：
\begin{equation}
p_i = \softmax(W h)_i, \quad \text{其中 } W \in \R^{V \times d_{\mathrm{model}}},\; h \in \R^{d_{\mathrm{model}}}
\end{equation}
Loss: $\loss = -\log p_y$（cross-entropy with target $y$）。

梯度对 logits $z = Wh$：
\begin{equation}
\frac{\partial \loss}{\partial z_j} = p_j - \mathbf{1}[j = y]
\end{equation}

梯度对 weight $W$：
\begin{equation}
\frac{\partial \loss}{\partial W_{j,:}} = (p_j - \mathbf{1}[j = y]) \cdot h^\top
\end{equation}

\textbf{为什么 LLM 不梯度消失？} 关键差异：

\begin{enumerate}[leftmargin=1.5em]
\item \textbf{LLM 的 softmax ``inputs''是可学习的 logits $z = Wh$（高维参数矩阵的输出），不是 mixing weights 作用于 fixed 高维向量}。

MoF 的计算图：$\ell \xrightarrow{\text{softmax}} \lambda \xrightarrow{\text{dot with } v^{(k)} \in \R^{65536}} \text{scalar outputs for loss}$

LLM 的计算图：$W, h \xrightarrow{\text{linear}} z \xrightarrow{\text{softmax}} p \xrightarrow{\text{CE loss}} \text{scalar}$

\item \textbf{LLM 中 softmax 的 ``选项'' ($V$ 个 vocabulary tokens) 彼此完全不同}。

对 LLM，$z_j$ 和 $z_k$（不同 token 的 logit）是\textbf{独立参数}的独立输出，没有 ``共享 base'' 的结构。梯度 $p_j - \mathbf{1}[j=y]$ 的量级 $\sim O(1)$（如 $p_y \approx 0.9$ 时梯度 $= -0.1$）。

对 MoF，$v^{(0)}$ 和 $v^{(1)}$ 共享 99\% 的内容（同一 base model + 小 LoRA）。Softmax 的 mean-subtraction 消去了共享部分，只留下 1\% 的 LoRA 差异。

\item \textbf{维度角色不同}：

LLM 中 softmax 作用于 \textbf{输出空间}（$V$ 个 token 概率），梯度直接是 $O(1)$ 标量信号 $(p_j - \mathbf{1}[j=y])$，不涉及高维 dot product。

MoF 中 softmax 作用于 \textbf{$K$ 个 teacher 的权重}（仅 2--3 个），然后乘以 $d = 65536$ 维的 teacher velocity，梯度需要通过 $\R^d$ 空间的 dot product 传递——方向对齐性 $\sim 1/\sqrt{d}$ 导致信号大量 cancellation。

\item \textbf{总结}：
\end{enumerate}

\begin{center}
\renewcommand{\arraystretch}{1.5}
\begin{tabular}{@{}p{3cm}p{5.5cm}p{5.5cm}@{}}
\toprule
& \textbf{LLM (lm\_head + softmax)} & \textbf{MoF (logits + softmax + velocity mix)} \\
\midrule
Softmax 选项数 & $V \sim 32000$--$128000$ & $K = 2$--$3$ \\
选项间差异 & $z_j, z_k$ 完全独立（无共享结构） & $v^{(j)}, v^{(k)}$ 共享 95--99\% (base model) \\
梯度 bottleneck & 无（$\partial\loss/\partial z_j = p_j - \mathbf{1}[j=y] \sim O(1)$） & $\mathbf{g}^\top(v^{(j)} - \bar{v}) \sim O(10^{-6})$（高维 dot product + LoRA diff） \\
高维 dot product & 无（loss 直接在概率空间定义） & 有（$\R^{65536}$ 维 velocity dot product） \\
梯度量级 & $\sim 0.01$--$1.0$ & $\sim 10^{-10}$ \\
\bottomrule
\end{tabular}
\end{center}

\textbf{根本区别}：LLM 的 loss（cross-entropy）直接在 softmax 输出概率空间上定义，梯度 $= p_j - \text{target}$ 是 $O(1)$ 标量。MoF 的 loss（NFT MSE）定义在 \textbf{softmax 输出之后的高维 velocity 空间}上，梯度需要从 $\R^d$ 投影回 $\R^K$ 的 logit 空间——这个投影（dot product + mean subtraction）造成了灾难性的信号衰减。

等价地说：\textbf{如果 LLM 的 loss 不是 cross-entropy（直接对 $p$ 求导），而是定义在 ``$\sum_i p_i \cdot \text{embedding}_i$'' 这样的 embedding mixture 上的 MSE，且所有 token embedding 高度相似（如同义词），那么 LLM 同样会梯度消失。}
\end{tcolorbox}

\subsection{Solution A: Direct Reward Loss（Contextual Bandit）}

\subsubsection{动机}
NFT loss 的梯度需经过高维 velocity 空间的 dot product，被 teacher 相似性压缩。直接在 logit 空间用 reward 信号更新权重，完全绕过 velocity dot product。

\subsubsection{Loss 定义}

\begin{definition}[Reward-Weighted Log-Likelihood Loss]
\label{def:reward_loss}
对 batch 中 prompt set $s$ 的 sample $x_b$，设 per-sample advantage 为 $\adv_b$：
\begin{equation}
\boxed{
\loss^{\mathrm{bandit}}_s = -\frac{1}{B_s \cdot T}\sum_{b \in \mathcal{B}_s}\sum_{i=1}^T \adv_b \cdot \log \lambda_{\hat{k}(b,i),\,i,\,s}
}
\label{eq:bandit_loss}
\end{equation}
其中 $\hat{k}(b,i)$ 是样本 $b$ 在 timestep $i$ 上的``oracle best teacher''：
\begin{equation}
\hat{k}(b,i) = \arg\max_k\; \mathbf{g}^{(b)}_{i,s} \cdot v^{(k)}_i \quad\text{（NFT gradient 方向上最优的 teacher）}
\end{equation}

\textbf{Soft 版本}（无需 argmax，更平滑）：
\begin{equation}
\boxed{
\loss^{\mathrm{soft}}_s = -\frac{1}{B_s \cdot T}\sum_{b \in \mathcal{B}_s}\sum_{i=1}^T \sum_k \alpha^{(b)}_{k,i}\, \adv_b \cdot \log \lambda_{k,i,s}
}
\label{eq:soft_bandit_loss}
\end{equation}
其中 $\alpha^{(b)}_{k,i}$ 是 teacher $k$ 在 sample $b$, timestep $i$ 上的``contribution indicator''（如 softmax of per-teacher rewards）。
\end{definition}

\subsubsection{梯度分析}

\begin{proposition}[Bandit Loss Gradient]
\label{prop:bandit_gradient}
对 soft bandit loss~\eqref{eq:soft_bandit_loss}，梯度为：
\begin{equation}
\boxed{
\frac{\partial \loss^{\mathrm{soft}}_s}{\partial \ell_{j,i,s}} = -\frac{1}{\tau\, B_s}\sum_{b \in \mathcal{B}_s} \adv_b \left[\alpha^{(b)}_{j,i} - \lambda_{j,i,s}\sum_k \alpha^{(b)}_{k,i}\right]
}
\label{eq:bandit_gradient}
\end{equation}

当 $\sum_k \alpha^{(b)}_{k,i} = 1$（probability assignment）时简化为：
\begin{equation}
\frac{\partial \loss^{\mathrm{soft}}_s}{\partial \ell_{j,i,s}} = -\frac{1}{\tau\, B_s}\sum_b \adv_b \left(\alpha^{(b)}_{j,i} - \lambda_{j,i,s}\right)
\label{eq:bandit_gradient_simple}
\end{equation}

\textbf{梯度量级}：$\sim \frac{|\adv|}{B_s \cdot \tau} \cdot |\alpha_j - \lambda_j| \sim 0.01$--$0.1$。
\end{proposition}

\begin{tcolorbox}[colback=green!5, colframe=green!50!black, title=关键优势]
Bandit gradient \eqref{eq:bandit_gradient_simple} 依赖于：
\begin{itemize}[leftmargin=1em]
\item $\adv_b$：advantage 量级 $\sim 0.2$--$2.0$
\item $\alpha_j - \lambda_j$：当前权重与``目标权重''的差异 $\sim 0.1$--$0.5$
\end{itemize}
\textbf{完全不依赖 teacher velocity 差异 $\|v^{(j)} - \bar{v}\|$}。

对比 NFT gradient \eqref{eq:grad_dot_product}：
\begin{center}
\renewcommand{\arraystretch}{1.3}
\begin{tabular}{lcc}
\toprule
& \textbf{NFT Gradient} & \textbf{Bandit Gradient} \\
\midrule
梯度量级 & $\sim 10^{-10}$ & $\sim 10^{-2}$ \\
依赖 teacher 相似性 & \ding{51}（正比于 $\|\delta^{(j)} - \bar{\delta}\|$） & \ding{55} \\
依赖 high-dim dot product & \ding{51}（$\cos\theta \sim 1/\sqrt{d}$） & \ding{55} \\
受 $\beta$ 影响 & \ding{55}（cancelled） & N/A \\
\bottomrule
\end{tabular}
\end{center}
\end{tcolorbox}

\subsubsection{实际实现：Per-Teacher Advantage}

需要为每个 teacher 获得独立的 advantage 信号。两种方案：

\paragraph{方案 A1: Counterfactual per-teacher reward.}
对每个 sample $b$，在采样阶段记录每个 teacher 的``贡献度''：
\begin{equation}
\alpha^{(b)}_{k,i} = \frac{\exp(\text{sim}(v^{(k)}_i, v^*_i))}{\sum_{k'} \exp(\text{sim}(v^{(k')}_i, v^*_i))}
\end{equation}
其中 $v^* = (x_t - x_0)/\sigma$ 是 ground truth velocity。$\alpha$ 衡量每个 teacher 在该 timestep 上与 ground truth 的对齐程度。

\paragraph{方案 A2: Per-teacher baseline evaluation.}
在每个 eval epoch 中对每个 teacher 单独生成图像并计算 reward：
\begin{equation}
\alpha^{(b)}_{k,i} = \begin{cases}
1 & \text{if } k = \arg\max_{k'} R_{\mathrm{in}(s)}(\text{img from teacher } k') \\
0 & \text{otherwise}
\end{cases}
\end{equation}

\subsection{Solution B: REINFORCE on Logit Space}

\subsubsection{Loss 定义}

\begin{definition}[REINFORCE Policy Gradient for Mixing Weights]
将 $\lambda_{k,i,s}$ 视为在每个 timestep 上``选择 teacher $k$'' 的概率，使用 policy gradient：
\begin{equation}
\boxed{
\nabla_{\ell_{:,i,s}} J_s = \E_{k \sim \lambda_{:,i,s}}\left[\adv_s \cdot \nabla_{\ell_{:,i,s}} \log \lambda_{k,i,s}\right]
}
\label{eq:reinforce}
\end{equation}

Soft relaxation（不采样，直接用期望）：
\begin{equation}
\loss^{\mathrm{PG}}_s = -\frac{1}{T}\sum_{i=1}^T \adv_s \cdot \sum_k \lambda_{k,i,s}\, \log \lambda_{k,i,s}
\label{eq:soft_pg}
\end{equation}
\end{definition}

\subsubsection{梯度分析}

\begin{proposition}[REINFORCE Gradient]
对 soft PG loss~\eqref{eq:soft_pg}：
\begin{equation}
\frac{\partial \loss^{\mathrm{PG}}_s}{\partial \ell_{j,i,s}} = -\frac{\adv_s}{\tau}\sum_k \lambda_{k,i,s}(\delta_{kj} - \lambda_{j,i,s})(\log\lambda_{k,i,s} + 1)
\label{eq:pg_gradient}
\end{equation}

简化（对 $K=2$, uniform $\lambda = 0.5$）：
\begin{equation}
\left|\frac{\partial \loss^{\mathrm{PG}}_s}{\partial \ell_{j,i,s}}\right| \sim \frac{|\adv_s|}{4\tau} \cdot |\log(0.5) + 1| = \frac{|\adv_s|}{4\tau} \cdot 0.307
\end{equation}

\textbf{量级}: $\sim |\adv_s| / 13 \sim 0.02$--$0.15$。同样不依赖 teacher velocity 差异。
\end{proposition}

\begin{remark}[REINFORCE 的局限性]
Soft PG loss~\eqref{eq:soft_pg} 本质上最大化 entropy $H(\lambda_{:,i,s})$ weighted by advantage。它推动权重向 high-advantage teacher 集中，但：
\begin{itemize}[leftmargin=1em]
\item 只有\textbf{一个} advantage $\adv_s$ per sample（不是 per-teacher advantage）。
\item 所有 timestep 共享同一 $\adv_s$ → 无法区分哪个 timestep 的哪个 teacher 贡献了高 reward。
\item 需要 per-teacher counterfactual reward 才能提供 per-$(k,i)$ 的 gradient signal。
\end{itemize}
\end{remark}

\subsection{Solution C: Residual-Space NFT}

\subsubsection{动机}
保持 NFT 框架，但在数值上放大 teacher 差异信号。将 velocity 分解为 base + residual，在 residual 空间操作。

\subsubsection{Loss 定义}

\begin{definition}[Residual-Space NFT]
显式计算 base velocity 和 per-teacher residuals：
\begin{align}
v_{\mathrm{base}} &= \frac{1}{K}\sum_k v^{(k)} \quad\text{（teacher 均值作为 base，detached）} \\
\delta^{(k)} &= v^{(k)} - v_{\mathrm{base}} \quad\text{（per-teacher residual）}
\end{align}

Combined velocity 重写为：
\begin{equation}
v^{\mathrm{new}}_{i,s} = v_{\mathrm{base}} + \sum_k \lambda_{k,i,s}\, \delta^{(k)}_i
\end{equation}

NFT loss 不变，但 loss gradient $\mathbf{g}$ 现在需与 $\delta^{(k)} - \bar{\delta}$ 做 dot product。
\end{definition}

\subsubsection{梯度分析}

\begin{proposition}[Residual-Space 梯度]
数学上，residual decomposition \textbf{不改变梯度}：
\begin{equation}
\frac{\partial \loss}{\partial \ell_{j,i,s}} = \frac{\lambda_j}{\tau}\,\mathbf{g}^\top\left(\delta^{(j)} - \bar{\delta}\right) = \frac{\lambda_j}{\tau}\,\mathbf{g}^\top\left(v^{(j)} - \bar{v}\right) \quad\text{（恒等）}
\end{equation}
因为 $v_{\mathrm{base}}$ 是常数，$\sum_k \lambda_k = 1$，base 项在 softmax Jacobian 中消去。

\textbf{但数值实现上可以有改进}：若在 fp32 下显式计算 $\delta^{(k)} = v^{(k)} - v_{\mathrm{base}}$ 并存储（避免在 bf16 backward 中隐式相减），可减少精度损失。
\end{proposition}

\begin{remark}[Partial Fix 的量级改善]
Residual-space 实现的主要收益：
\begin{itemize}[leftmargin=1em]
\item 避免 bf16 下隐式计算 $v^{(0)} - v^{(1)}$（大数减大数的精度损失）
\item 但\textbf{数学上梯度量级不变}——Theorem~\ref{thm:fundamental_limit} 的约束无法绕过
\item 预期改善：$\sim 2$--$10\times$（bf16 precision recovery），不足以解决 $10^8\times$ 的差距
\end{itemize}
\end{remark}

\subsection{Solution D: Unnormalized Additive Mixing}

\subsubsection{动机}
打破 Theorem~\ref{thm:fundamental_limit} 的约束 $\sum w_k = 1$。使用不归一化的 per-teacher scaling，使梯度正比于 $\delta^{(k)}$ 本身而非 $\delta^{(k)} - \bar{\delta}$。

\subsubsection{Loss 定义}

\begin{definition}[Unnormalized Additive Velocity Mixing]
\begin{equation}
\boxed{
v^{\mathrm{new}}_{i,s} = v_{\mathrm{base}}(x, t_i) + \sum_{k=1}^K \alpha_{k,i,s}\, \delta^{(k)}(x, t_i)
}
\label{eq:unnormalized_mix}
\end{equation}
其中 $\alpha_{k,i,s} = \ell_{k,i,s}$（直接使用 logit，无 softmax），$v_{\mathrm{base}} = v^{(\mathrm{ref})}$ 是参考模型（base model 或 teacher 均值），$\delta^{(k)} = v^{(k)} - v_{\mathrm{base}}$。

Loss 仍为 NFT loss，但以 $v^{\mathrm{new}}$ 的新定义代入。
\end{definition}

\subsubsection{梯度分析}

\begin{proposition}[Unnormalized Gradient]
\begin{equation}
\boxed{
\frac{\partial \loss}{\partial \ell_{j,i,s}} = \mathbf{g}_{i,s}^\top\, \delta^{(j)}_i
}
\label{eq:unnorm_gradient}
\end{equation}

\textbf{无 mean subtraction}！梯度直接为 loss gradient $\mathbf{g}$ 与 LoRA residual $\delta^{(j)}$ 的 dot product。

量级估计：
\begin{align}
\left|\frac{\partial \loss}{\partial \ell_{j,i,s}}\right| &\approx \|\mathbf{g}\| \cdot \|\delta^{(j)}\| \cdot |\cos\phi| \\
&\sim 2.6 \times 10^{-4} \cdot 2.6 \cdot 0.004 \\
&\sim 2.7 \times 10^{-6} \quad\text{（per timestep, before GAS）}
\end{align}
After GAS: $\sim 3.4 \times 10^{-8}$. Grad norm: $\sim 10^{-7}$.

\textbf{改善}: $\sim 10^3\times$ vs NFT's $10^{-10}$，因为去掉了 $(\delta^{(j)} - \bar{\delta}) \to \delta^{(j)}$ 的 mean subtraction 因子。
\end{proposition}

\begin{remark}[Unnormalized Mixing 的代价]
\begin{itemize}[leftmargin=1em]
\item Combined velocity 不再是 teacher 的凸组合 → 可能超出 teacher velocity 的凸包
\item 需要正则化约束 $\alpha_k$ 的范围（如 $L_2$ penalty 或 clip）
\item Level~1 保证（$\geq$ single teacher）不再自动成立——需额外约束 $\sum_k \alpha_k \approx 1$
\item 实际收益有限（$\sim 10^{3}\times$ 改善仍不如 Solution A 的 $\sim 10^{8}\times$）
\end{itemize}
\end{remark}

\subsection{推荐方案}

\begin{tcolorbox}[colback=blue!5, colframe=blue!50!black, title=推荐：Solution A (Direct Reward Loss)]
\textbf{核心思路}：不通过高维 velocity 空间传递梯度，直接用 reward/advantage 信号在 logit 空间做梯度更新。

\textbf{实现方式}：
\begin{enumerate}[leftmargin=1.5em]
\item 保持 sampling/inference 不变（仍用 softmax combined velocity 生成图像）
\item 计算 reward 和 advantage 不变
\item 替换 optimize 中的 NFT loss 为：
\begin{equation}
\loss = -\frac{1}{B_s T}\sum_b\sum_i \adv_b \cdot \log\lambda_{\hat{k}(b,i), i, s}
\end{equation}
\item 其中 $\hat{k}(b,i)$ 可简化为``in-domain teacher index''（teacher-biased REINFORCE），或由 per-teacher reward evaluation 确定
\end{enumerate}

\textbf{优势}：
\begin{itemize}[leftmargin=1em]
\item 梯度量级 $\sim 10^{-2}$（$\sim 10^8\times$ improvement）
\item 完全不依赖 teacher velocity 相似性
\item 与 NFT 框架正交——可作为 ``warm start'' 后切换到 NFT 的策略
\item 保持 Level-1 guarantee（teacher-biased init + reward-driven update only increases in-domain）
\end{itemize}

\textbf{潜在问题}：
\begin{itemize}[leftmargin=1em]
\item Per-timestep teacher attribution ($\hat{k}(b,i)$ or $\alpha_{k,i}$) 需要额外计算
\item 若只有 per-sample advantage（不是 per-$(k,i)$ advantage），所有 timestep 收到相同信号 → 学不到 timestep-specific 策略
\item 可能需要与 NFT loss 混合使用（NFT 提供 per-timestep direction，bandit 提供 magnitude）
\end{itemize}
\end{tcolorbox}
